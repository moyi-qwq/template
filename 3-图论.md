### DIJKSTRA

```cpp
int n, m, s;
vector<pair<int, int>> e[maxn]; // v, w 到达点 长度
int dis[maxn]; //距离
bool vis[maxn]; //是否访问过
void Dijk() {
    mst(vis);
    for(int i = 0; i <= n +7; i++) dis[i] = inf;
    dis[s] = 0;
    priority_queue<pair<int, int>> q; // u,dis
    q.push(make_pair(s,0));
    while(!q.empty()) {
        int u = q.top().first, d = q.top().second;
        q.pop();
        for(auto y : e[u]) {
            int v = y.first, w = y.second;
            int td = d + w;
            if(td < dis[v]) {
                dis[v] = td;
                q.push(make_pair(v,td));
            }
        }
    }
}
```

### SPFA+判负环

```
int n, m, s;
vector<pair<int, int>> e[maxn]; //to w 边终点，边权
int dis[maxn], cnt[maxn], vis[maxn]; // 0-(n-1)

bool spfa(int s) { // 返回true如果没有负环 否则返回false
    for(int i = 0; i < n; i++) {
        dis[i] = inf;
    }
    queue<int> q;
    dis[s] = 0; vis[s] = 1;
    q.push(s);
    while(!q.empty()) {
        int u = q.front();
        q.pop(); vis[u] = 0;
        for(auto y : e[u]) {
            int v = y.first, w = y.second;
            if(dis[v] > dis[u] + w) {
                dis[v] = dis[u] + w;
                cnt[v] = cnt[u] + 1;  // 记录最短路经过的边数
                if (cnt[v] >= n) return false;
                // 在不经过负环的情况下，最短路至多经过 n - 1 条边
                // 因此如果经过了多于 n 条边，一定说明经过了负环
                if (!vis[v]) q.push(v), vis[v] = 1;
            }
        }
    }
    return true;
}
```

### 倍增

输入N，M，S：树的结点个数、询问的个数和树根结点的序号。

接下来N-1行是边，M行询问最近公共祖先是谁。


```cpp
int N, M, S;
int fa[maxn][31], dep[maxn];
//fa[i][j]：第i个点的第2^j个祖先
vector<int> G[maxn];

void dfs(int root,int fno) {
    fa[root][0] = fno;
    dep[root] = dep[fno] + 1;
    //初始化fa
    for(int i = 1; i < 31; ++i) {
        fa[root][i] = fa[fa[root][i-1]][i-1];
    }
    //遍历
    for(auto y : G[root]) {
        if(y == fno) continue;
        dfs(y, root);
    }
}

int lca(int x, int y) {
    if(dep[x] > dep[y]) swap(x,y);
    int tem = dep[y] - dep[x];
    for(int i = 0; tem; ++i, tem >>= 1)
    {
        if(tem&1) y = fa[y][i];
    }
    if(y == x) return x;
    for(int i = 30; i >= 0 && y != x; --i) {
        if(fa[x][i] != fa[y][i]) {
            x = fa[x][i];
            y = fa[y][i];
        }
    }
    return fa[x][0];
}

int main() {
    cin>>N>>M>>S;
    for(int i = 1; i < N; i++) {
        int x, y;
        cin>>x>>y;
        G[x].push_back(y);
        G[y].push_back(x);
    }
    dfs(S, 0);
    for(int i = 1; i <= M; i++) {
        int x, y;
        cin>>x>>y;
        cout<<lca(x,y)<<endl;
    }
    return (0);
}
```

## 欧拉路径/回路

判别法：

+ 对于无向图G，G是欧拉图当且仅当G是连通的且没有奇度顶点。(欧拉回路
+ 对于无向图G，G是半欧拉图当且仅当G是连通的且G中恰有 个或 个奇度顶点。(欧拉路径
+ 对于有向图G，G是欧拉图当且仅当G的所有顶点属于同一个强连通分量且每个顶点的入度和出度相同。(欧拉回路
+ 对于有向图G，G是半欧拉图当且仅当：(欧拉路径

	+ 如果将G中的所有有向边退化为无向边时，那么G的所有顶点属于同一个连通分量。
	+ 最多只有一个顶点的出度与入度差为1。
	+ 最多只有一个顶点的入度与出度差为1。
	+ 所有其他顶点的入度和出度相同。

题：

```
无向图找欧拉路径,输出字典序最小的路径。
```

```cpp
int n, m;
int e[maxn][maxn], du[maxn]; //邻接矩阵 点的度
stack<int> ans; //访问点的顺序

void dfs(int u) {
    for(int i = 1; i <= n; i++) {
        if(e[u][i]) {
            e[u][i]--; e[i][u]--;
            dfs(i);
        }
    }
    ans.push(u);
}

signed main() {
    cin>>m;
    for(int i = 1; i <= m; i++) {
        int u, v; cin>>u>>v;
        n = max(n, u); n = max(n, v);
        e[u][v]++;
        e[v][u]++;
        du[u]++; du[v]++;
    }
    int s = 1;
    for(int i = 1; i <= n; i++) {
        if(du[i]%2==1) {s = i; break;}
    }
    dfs(s);
    while(ans.size()) {
        cout<<ans.top()<<endl;
        ans.pop();
    }
    return (0);
}
```

## 拓扑排序
```cpp
int n, m;
int deg[maxn]; //入度
vector<int> e[maxn];
vector<int> ans;

bool toposort() { // 0-indexed
    queue<int> q;
    for(int i = 0; i < n; i++) {
        if(deg[i]==0) q.push(i);
    }
    while(!q.empty()) {
        int u = q.front();
        q.pop();
        ans.push_back(u);
        for(auto y : e[u]) {
            --deg[y];
            if(deg[y]==0) q.push(y);
        }
    }
    if(ans.size()==n) return true;
    else return false;
}
```

### 割点和割边（无向图

tarjan同时处理割点和桥

```
int n, m;
vector<int> e[maxn];
int dfn[maxn], low[maxn], vis[maxn];
//访问顺序 不过fa最低访问dfn 记录
int iscut[maxn];
int isbridge[maxn], father[maxn];

void tarjan(int u, int fa, int dep) { //割点割边同时处理
    father[u] = fa;
    vis[u] = 1;
    int child = 0;
    low[u] = dfn[u] = dep;
    for(auto v : e[u]) {
        if(v != fa && vis[v] == 1) {
            low[u] = min(low[u], dfn[v]);
        }
        if(vis[v] == 0) {
            tarjan(v, u, dep + 1);
            child++;
            low[u] = min(low[u], low[v]);
            if( (fa==-1 && child>1) || (fa!=-1 && low[v] >= dfn[u])) {
                iscut[u] = true;
            }
            if(low[v] > dfn[u]) {
                isbridge[v] = true; //割边 会重复所以要isbridge[]记录
            }
        }
    }
    vis[u] = 2;
}
```

## 最小生成树

```cpp
//唯一
struct edge{
    int from, to, val;
}e[maxn];
int n, m, ans;
int fa[maxn];
bool cmp(edge a, edge b){return a.val < b.val;}
void init()
{
    for(int i = 0; i <= n; i++)
    {
        fa[i] = i;
    }
    ans = 0;
}
int find(int x) {return fa[x]==x?x:(fa[x] = find(fa[x]));}
bool uniKruskal()
{
    int sum1 = 0, sum2 = 0;//已使用的 可能使用的
    int p = 0;//相同指针
    int flag = 0, num = 0;
    for(int i = 1; i <= m + 1; i++)
    {
        if(p<i)
        {
            if(sum1 != sum2)
            {
                flag = 1;
                break;
            }
            sum1 = 0, sum2 = 0;
            for(int j = i; j <= m+1; j++)
            {
                if(e[j].val != e[i].val)
                {
                    p = j-1;
                    break;
                }
                if(find(e[j].from) != find(e[j].to))
                ++sum2;
            }
        }
        if(i>m) break;
        int x = find(e[i].from);
        int y = find(e[i].to);
        if(x != y && num != n-1)
        {
            num++;
            sum1++;
            // merge(x, y);
            fa[x] = fa[y];
            ans += e[i].val;
        }
    }
    if(flag) return false;
    else return true;
}

int main()
{
    freopen("D:/c++source file/intxt/in.txt","r",stdin);
    // ios :: sync_with_stdio(0);
    // cin.tie(0);

    {
```
